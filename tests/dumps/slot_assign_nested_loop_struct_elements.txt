struct Pair { x: integer, y: integer }

fn sum_matching(pairs: vector<Pair>, xs: vector<Pair>) -> integer {
    total = 0;
    for p in pairs {
        for q in xs {
            if p.x == q.x { total = total + p.y }
        }
    };
    total
}
